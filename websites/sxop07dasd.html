def character_shift(text, shift):
    shifted_text = ""
    for char in text:
        if char.isalpha():
            shifted = chr(((ord(char) - 65 + shift) % 26) + 65) if char.isupper() else chr(((ord(char) - 97 + shift) % 26) + 97)
            shifted_text += shifted
        else:
            shifted_text += char
    return shifted_text
